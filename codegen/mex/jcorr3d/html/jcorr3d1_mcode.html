1% JCORR3D.M
 2% This function performs a correlation on a 3D volume with a 3D kernel. It
 3% is faster than MATLAB's convn or imfilter, and produces the same result 
 4% (Relative Error is of order 1E-12).
 5% The drawback is that for large correlations out-of-memory errors might occur,
 6% this is solved by splitting the volume which is convoluted.
 7%
 8% Written and (c) by:   Joost Weijs, University of Twente, The Netherlands
 9%
10%    USAGE:
11%
12%       out = jconv3d ( dataset , kernel , split )
13%
14%    INPUTS:
15%       dataset: A 3-dimensional dataset to be convoluted with the mirrored kernel.
16%       Type can be single or double.
17%
18%       kernel: A 3-dimensional kernel with which the dataset is
19%       convoluted. Can also be single or double precision. The kernel is
20%       mirrored to perform the correlation.
21%
22%       split: An integer value. The dataset volume is splitted in split
23%       (more or less) equal parts in the z-direction. Using the
24%       overlap-add method the convolution of the complete dataset will be
25%       calculated. If you encounter OUT OF MEMORY errors while running
26%       this function, you might want to increase this value. The cost of
27%       higher values are slightly increased computation times.
28%
29%     OUTPUT:
30%       out: The result of the correlation
31%
32%       (c) 2008 Joost Weijs, University of Twente, The Netherlands
33
34
35function out=jcorr3d(dataset,kernel,split)
36
37    sizedataset=size(dataset);
38    sizekernel=size(kernel);
39
40    %invert kernel
41    kernel=kernel(sizekernel(1):-1:1,sizekernel(2):-1:1,sizekernel(3):-1:1);
42    C=zeros(sizedataset+sizekernel-[1,1,1]);
43    z_start=1;
44    i=0;
45    while z_start<sizedataset(3)
46        i=i+1;
47        z_end=min([ceil(i*sizedataset(3)/split),sizedataset(3)]);
48        A=zeros(sizedataset(1)+sizekernel(1)-1,sizedataset(2)+sizekernel(2)-1,z_end-z_start+sizekernel(3));
49        B=A;
50        A(1:sizedataset(1),1:sizedataset(2),1:z_end-z_start+1)   = dataset(:,:,z_start:z_end);    
51        B(1:sizekernel(1) ,1:sizekernel(2) ,1:sizekernel(3))     = kernel;
52        fftA = fftn(A);
53        fftB = fftn(B);
54        fftC = fftA.*fftB;
55        Ct   = ifftn(fftC);
56        C(:,:,z_start:z_end+sizekernel(3)-1)   = C(:,:,z_start:z_end+sizekernel(3)-1) + Ct;
57        z_start=z_end+1;
58        
59    end
60    out=C;
61%    sC=size(C);
62%    C(sC(1):-1:sC(1)-floor(sizekernel(1)/2),:,:)=[]; C(1:ceil(sizekernel(1)/2),:,:)=[];  
63%    C(:,sC(2):-1:sC(2)-floor(sizekernel(2)/2),:)=[]; C(:,1:ceil(sizekernel(2)/2),:)=[];  
64%    C(:,:,sC(3):-1:sC(3)-floor(sizekernel(3)/2))=[]; C(:,:,1:ceil(sizekernel(3)/2))=[];  
65% %imagesc(C(:,:,20))
66%     out=C( floor(sizekernel(1)/2) : ceil(sC(1) - sizekernel(1)/2) ...
67%         ,  floor(sizekernel(2)/2) : ceil(sC(2) - sizekernel(2)/2) ...
68%         ,  floor(sizekernel(3)/2) : ceil(sC(3) - sizekernel(3)/2)  );
69
70end
